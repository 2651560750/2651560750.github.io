多线程学习之实现方法二：自定义继承Theard类

2021-03-29 2 min read 
package com.deng.pojo;
//创建线程方式二，实现runnable接口，重写run方法，执行线程要丢入runnable接口实现类，调用start方法
public class TestThread3 implements Runnable{
    @Override
    public void run()
    {
        //run方法线程体
        for (int i=1;i<=20;i++)
        {
            System.out.println("我在看代码--"+i);
        }
    }

    public static void main(String[] args) {
        //main线程，主线程
        //创建一个线程对象
        TestThread1 testThread1=new TestThread1();
        //将实现runnable接口的类丢到Theard线程类中执行
        //new Thread(testThread1).start();
Thread thread=new Thread(testThread1);
//调用START方法开启线程
thread.start();
        for (int i=1;i<=2000;i++)
        {
            System.out.println("我在看多线程--"+i);
        }
    }
}

#演示：龟兔赛跑

package com.deng.pojo;

public class race implements Runnable {
    private static String winner;

    @Override
    public void run() {
        for (int i = 1; i <= 100; i++) {
            if(Thread.currentThread().getName().equals("邓贝")&& i%10==0){
                try {
                    Thread.sleep(1);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
   boolean flag=finsh(i);
   if (flag)
   {
       break;
   }
            System.out.println(Thread.currentThread().getName()+"跑了"+i+"步");
        }
    }
   private boolean finsh(int step)
   {
       if(winner!=null)
       {
           return true;
       }
       if(step==100)
       {
           winner=Thread.currentThread().getName();
           System.out.println("winner is:"+winner);
           return true;
       }
       return false;
   }

    public static void main(String[] args) {
        race r=new race();
        new Thread(r,"岁东航").start();
        new Thread(r,"邓贝").start();

    }
}
