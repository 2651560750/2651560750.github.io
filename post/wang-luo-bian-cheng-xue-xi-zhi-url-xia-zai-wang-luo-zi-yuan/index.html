2021-04-07 1 min read 
package com.deng.URL01;

import java.io.FileOutputStream;
import java.io.InputStream;
import java.net.HttpURLConnection;
import java.net.URL;

public class Urldown {
    public static void main(String[] args) throws Exception {
        URL url=new URL("https://th.bing.com/th/id/Rde1bdc82f8063c8df1b62a686c12042d?rik=PNiygsObQkS7hw&riu=http%3a%2f%2fimage.photocnc.com%2fphotocnc%2f2020-06%2f29%2f202006290131193f0894ee4d24494d61d50a61a3c0ef49.jpg.w1680.jpg&ehk=vlDDobgu9MW4%2f9CP8wP%2bTxuJ5%2fJ6S7HY%2faTAUy1%2beaQ%3d&risl=&pid=ImgRaw");
        HttpURLConnection urlConnection=(HttpURLConnection) url.openConnection();
        InputStream inputStream=urlConnection.getInputStream();
        FileOutputStream fos=new FileOutputStream("yilishabais.jpg");
        byte[] buffer=new byte[1024];
        int len;
        while ((len=inputStream.read(buffer))!=-1)
        {
            fos.write(buffer,0,len);
        }
        fos.close();
        inputStream.close();
        urlConnection.disconnect();
    }
}
