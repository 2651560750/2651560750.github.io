#函数的二种定义方式
第一种，可以使用function关键字来定义函数。
使用的格式如下:
function函数名(形参列表)
函数体
}
在JavaScript语言中，如何定义带有返回值的函数?
只需要在函数体内直接使用return语句返回值即可!

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <script type="text/javascript">
/*        第一种，可以使用function关键字来定义函数。
使用的格式如下:
    function函数名(形参列表)
        函数体
        }*/
//定义一个无惨函数
function fun(){
    alert("岁东航也是个王八犊子操");
}
//调用函数
fun();
//定义一个有参函数
function fun2( a,b){
    alert("岁东航他妈的拉了"+a+"坨屎,他直接说"+b);
}
fun2(12,"曹尼玛");
//带有返回值类型的函数
function fun3(sum1,sum2){
    var sum =sum1+sum2
    return sum;
}
alert(fun3(25,65));
    </script>
</head>
